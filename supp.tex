\begin{baselinebox}
    \textbf{Prompt 6. Error Taxonomy Judge Prompt (Claude Opus 4.6 via AWS Bedrock)}

    \vspace{1em}
    You are an expert medical diagnostician classifying errors in a clinical diagnosis task.

    \vspace{0.5em}
    A model was given a patient case and a list of differential diagnoses, and asked to select the correct diagnosis verbatim from the list. The model's prediction was scored INCORRECT under exact text matching.

    \vspace{0.5em}
    Your task: classify this error into exactly one of five categories.

    \vspace{0.5em}
    \textbf{Categories:}
    \begin{enumerate}[label=(\alph*), leftmargin=*, noitemsep, topsep=2pt]
        \item \textbf{SURFACE\_MISMATCH} --- The prediction is clinically correct but differs in surface form (synonym, eponym, abbreviation, nomenclature variant, or minor wording difference). The model identified the right condition.
        \item \textbf{SAME\_FAMILY} --- The prediction names a condition in the same disease family or anatomical system as the ground truth, but is the wrong specific entity.
        \item \textbf{PLAUSIBLE\_DIFFERENTIAL} --- The prediction is a plausible differential diagnosis that was on the provided list, but is incorrect. The model's reasoning was clinically defensible but wrong.
        \item \textbf{OFF\_TOPIC} --- The prediction names a condition that is not on the differential list, is anatomically/pathologically unrelated, or is a hallucinated entity.
        \item \textbf{EMPTY\_OR\_REFUSAL} --- The prediction is empty, a refusal to answer, a parse failure, or contains no identifiable diagnosis.
    \end{enumerate}

    \vspace{0.5em}
    \textbf{Case information:}
    \\
    Ground truth diagnosis: \textit{[ground\_truth]}
    \\
    Model prediction: \textit{[prediction]}
    \\
    Differential diagnosis list: \textit{[differentials]}

    \vspace{0.5em}
    \textbf{Instructions:}
    Respond with ONLY the category letter (a, b, c, d, or e) on the first line, followed by a one-sentence justification on the second line.
\end{baselinebox}

\begin{algorithm}[H] % [H] is required to prevent floating
\caption{Inference protocol for robust diagnostic reasoning.
The algorithm employs diverse group beam search to generate multiple independent reasoning paths ($B$ beams, $G$ groups), followed by majority-vote aggregation to mitigate hallucination and select the most consensus-driven diagnosis.}
\label{alg:inference_workflow}
\begin{algorithmic}[1]
\Require 
    \State $\mathcal{C}$: Patient Case (History and Imaging Findings)
    \State $\mathcal{L}$: List of Differential Diagnoses $\{l_1, l_2, ..., l_n\}$
    \State $\mathcal{M}$: Fine-tuned On-device Model
    \State Parameters: Number of beams $B$, Number of groups $G$, Diversity penalty $\lambda$
\Ensure 
    \State $\hat{y}$: Final Selected Diagnosis
    \State $R$: Generated Reasoning Trace associated with $\hat{y}$

\Statex
\State \textbf{Prompt Construction}
\State $P \leftarrow$ \Call{FormatPrompt}{$\mathcal{C}, \mathcal{L}$} \Comment{Inject case and constraints into template}
\State $T_{in} \leftarrow$ \Call{Tokenize}{$P$}

\Statex
\State \textbf{Diverse Generation (Exploration)}
\State {Generate $B$ distinct reasoning paths using Group Beam Search}
\State $\mathcal{S} \leftarrow$ \Call{Generate}{$T_{in}, \mathcal{M}, \text{beams}=B, \text{groups}=G, \text{penalty}=\lambda$}
\State \Comment{Return sequences $\{s_1, s_2, .., s_B\}$ sorted by beam index}

\Statex
\State \textbf{Answer Extraction and Normalization}
\State $\mathcal{D}_{raw} \leftarrow \emptyset$ \Comment{List to store extracted answers}
\For{$i \leftarrow 1$ to $B$}
    \State $r_i, d_i \leftarrow$ \Call{ParseOutput}{$s_i$} \Comment{Split reasoning ($r$) and diagnosis ($d$) via delimiter}
    \State $d_i^{norm} \leftarrow$ \Call{NormalizeText}{$d_i$} \Comment{Lowercase, strip punctuation/whitespace}
    \State $\mathcal{D}_{raw} \leftarrow \mathcal{D}_{raw} \cup \{d_i^{norm}\}$
\EndFor

\Statex
\State \textbf{Majority Vote Aggregation}
\State $V \leftarrow$ \Call{CountFrequencies}{$\mathcal{D}_{raw}$} \Comment{Map unique diagnoses to vote counts}
\State $v_{max} \leftarrow \max(V)$
\State $\mathcal{T} \leftarrow \{d \mid V[d] = v_{max}\}$ \Comment{Set of diagnoses with the highest vote count}

\Statex
\State \textbf{Tie-Breaking and Final Selection}
\If{$|\mathcal{T}| = 1$}
    \State $\hat{y} \leftarrow \text{element in } \mathcal{T}$
\Else
    \State \Comment{Tie-breaker: Select the diagnosis appearing in the earliest beam index}
    \State $\hat{y} \leftarrow$ first $d_i^{norm} \in \mathcal{D}_{raw}$ such that $d_i^{norm} \in \mathcal{T}$
\EndIf

\State $idx \leftarrow$ index of first occurrence of $\hat{y}$ in $\mathcal{D}_{raw}$
\State \Return $\hat{y}, s_{idx}$

\end{algorithmic}
\end{algorithm}

\section*{Case Studies}
The following case studies illustrate the chain-of-thought reasoning capabilities of two fine-tuned models---gpt-oss-20b and Qwen3.5-35B---each fine-tuned on the Eurorad radiology dataset. For each model, we present paired before-and-after comparisons across two clinically diverse cases. Each case demonstrates the model's systematic approach to differential diagnosis: (1) identifying core clinical-radiologic patterns, (2) mapping each differential to observed features, (3) systematically eliminating unlikely diagnoses, and (4) converging on a final answer with supporting rationale.

\vspace{1em}

% gpt-oss-20b Cases
% -------------------------------------------------------
The following cases demonstrate the reasoning behaviour of gpt-oss-20b before and after fine-tuning. The base model responses reflect common failure modes including superficial elimination, circular reasoning, and incorrect final answers.

\begin{casebox}
    \textbf{Case Study 1: Primary Cardiac Lymphoma (gpt-oss-20b)}

    \vspace{1em}
    \textbf{Case Presentation}
    
    \textbf{Clinical History:} An 81-year-old male with paroxysmal atrial fibrillation presents with one-week onset of orthopnoea and exertional dyspnoea. Elevated troponins, third-degree AV block on ECG, and right ventricular thickening with pericardial effusion on echocardiogram.
    
    \vspace{0.5em}
    \textbf{Imaging Findings:} PET-CT reveals a large hypermetabolic mass through the right chambers sparing the left ventricle, and a hypermetabolic left lung base nodule. Chest CT confirms an infiltrative hypodense myocardial mass involving the right atrium, right ventricle, interatrial septum, and AV groove, with a cuff around the right coronary artery, moderate pericardial effusion, and a solid left lower lobe nodule biopsied under CT guidance.
    
    \vspace{0.5em}
    \textbf{Differentials:} Angiosarcoma, IgG4-related disease, Primary cardiac lymphoma, Erdheim-Chester disease, Undifferentiated pleomorphic sarcoma \quad \textbf{Ground Truth:} Primary cardiac lymphoma
    
    \vspace{1em}
    \hrule
    \vspace{1em}

    \textbf{Base Model Response (Before Fine-Tuning)}

    \vspace{0.5em}
    Angiosarcoma: Most common primary cardiac sarcoma. Usually involves right atrium, pericardial effusion, conduction block. Age typically younger (median 40--50). But can occur in older. Lung metastasis possible. So fits. \textit{[Issue: Makes excuses instead of using age as exclusion criterion; prematurely accepts this diagnosis]}

    \vspace{0.5em}
    IgG4-related disease: Can involve heart. Rarely presents as large infiltrative myocardial mass. Not typical. \textit{[Issue: Rambling, uncertain reasoning]}

    \vspace{0.5em}
    Primary cardiac lymphoma: Rare, involves right atrium, can cause conduction block, pericardial effusion. Could present with AV block. So possible. \textit{[Issue: Doesn't recognise AV block as KEY feature for lymphoma]}

    \vspace{0.5em}
    Erdheim-Chester disease: Osteosclerosis of long bones not mentioned. Less likely. \textit{[Issue: Superficial elimination]}

    \vspace{0.5em}
    Angiosarcoma is classic for right atrium, pericardial effusion, conduction block, metastasis to lung. Left ventricle is spared. That fits. The lung nodule suggests metastasis. So angiosarcoma is likely. \textit{[Issue: Wrong conclusion---prioritised lung nodule over AV block and hypodense CT pattern]}

    \noindent\textbf{Base Model Final Answer:} \colorbox{red!30}{Angiosarcoma} -- \textbf{INCORRECT}

    \vspace{0.5em}
    \noindent\textbf{Step 2 -- Differential mapping}

    \end{itemize}

    \noindent\textbf{Step 4 -- Final diagnosis}

        \item \textbf{Imaging precision:} Hypodense (lymphoma) vs.\ hyperdense haemorrhagic (angiosarcoma) distinction applied correctly
